All Options 

new WinBox(document.body, {

    title: "All Options",
    id: "my-window",
    class: "my-theme",
    color: "#ff005d",
    border: 4,
    width: 200,
    height: 200,
    x: 100,
    y: 100,
    top: 50,
    right: 50,
    bottom: 50,
    left: 50,
    html: "width: 200, height: 200",
    onfocus: function(){
        this.setColor("#ff005d"); // this => WinBox
    },
    onblur: function(){
        this.setColor("#999");
    },
    onresize: function(width, height){
        this.body.textContent = (
            "width: " + width + ", " +
            "height: " + height
        );
    },
    onmove: function(x, y){
        this.body.textContent = (
            "x: " + x + ", " +
            "y: " + y
        );
    },
    onclose: function(){
        alert("event 'onclose' fired")
    }
});
Run Code



Controls 

var winbox = new WinBox({
    title: "Controls",
    mount: document.getElementById("controls")
});
const buttons = {

    "minimize": function(){

        winbox.minimize();
    },
    "maximize": function(){

        winbox.maximize();
    },
    "fullscreen": function(){

        winbox.fullscreen();
    },
    "move": function(){

        winbox.move("center", "center");
    },
    "resize": function(){

        winbox.resize("50%", "50%");
    },
    "title": function(){

        winbox.setTitle("Title-" + Math.random());
    },
    "color": function(){

        winbox.setColor(
            "rgb(" +(Math.random() * 255 | 0) + "," +
                    (Math.random() * 255 | 0) + "," +
                    (Math.random() * 255 | 0) + ")"
        );
    },
    "close": function(){

        winbox.close();
    }
};
Run Code
